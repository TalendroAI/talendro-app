  const runDiagnostics = async () => {

    // 1. Check what Affinda actually parsed
    console.log('📋 STEP 1: Checking Affinda Parse Results');

      });

    try {
      const response = await fetch('/api/profile/draft');
      console.log('Server response status:', response.status);
      console.log('Server response headers:', Object.fromEntries(response.headers.entries()));
      
      if (response.ok) {
        const serverData = await response.json();
        console.log('Server draft data:', JSON.stringify(serverData, null, 2));
        console.log('Server LinkedIn fields:', Object.entries(serverData || {})
          .filter(([key, value]) => key.toLowerCase().includes('linkedin')));
      } else {
        const errorText = await response.text();
        console.log('Server error response:', errorText);
      }
    } catch (error) {
      console.error('Server fetch error:', error);
    }
    
    // 3. Check getStep3Prefill function
    console.log('⚙️ STEP 3: Testing getStep3Prefill Function');
    try {
      const prefillData = getStep3Prefill();

      console.log('Prefill LinkedIn fields:', Object.entries(prefillData || {})
        .filter(([key, value]) => key.toLowerCase().includes('linkedin')));
    } catch (error) {
      console.error('getStep3Prefill error:', error);
    }
    
    // 4. Check safeMergeResumeData function
    console.log('🔄 STEP 4: Testing safeMergeResumeData');
    const testData = { linkedin: 'https://linkedin.com/in/test', currentJobTitle: 'Test Title' };

      safeMergeResumeData(testData);
      const afterMerge = JSON.parse(localStorage.getItem('resumeData') || 'null');
      console.log('localStorage after merge:', JSON.stringify(afterMerge, null, 2));
    } catch (error) {
      console.error('safeMergeResumeData error:', error);
    }
