### A. Canonical step-3 prefill (server → client)

**File:** `server/mapping/affindaToCanonical.js` (excerpt for Step-3)

```js
const arrayify = (x) => Array.isArray(x) ? x : (x ? Object.values(x) : []);

export function affindaToStep3Prefill(raw) {
  const d = raw?.data || raw || {};

  // --- name ---
  const nameItem = arrayify(d.candidateName)[0];
  const first = nameItem?.parsed?.firstName?.parsed || nameItem?.parsed?.firstName?.raw || '';
  const last  = nameItem?.parsed?.familyName?.parsed || nameItem?.parsed?.familyName?.raw || '';
  const full  = [first, last].filter(Boolean).join(' ').trim();

  // --- email / phone ---
  const email = d?.email?.listItem?.[0]?.parsed || d?.email?.listItem?.[0]?.value || d?.emails?.[0] || '';
  const phone = d?.phoneNumber?.listItem?.[0]?.formattedNumber || d?.phoneNumber?.[0]?.rawText || '';

  // --- LinkedIn (any variant Affinda emits) ---
  const websites = arrayify(d.website) || arrayify(d.websites) || [];
  const social   = arrayify(d.socialMedia);
  const liFromWeb = websites
    .map(w => w?.parsed?.url || w?.url || w?.raw)
    .find(u => typeof u === 'string' && u.toLowerCase().includes('linkedin.com')) || '';

  const liFromSocial = social
    .map(s => s?.url || s?.parsed?.url || s?.handle)
    .find(u => typeof u === 'string' && u.toLowerCase().includes('linkedin.com')) || '';

  const linkedinUrl = liFromWeb || liFromSocial || '';

  // --- location (fill what exists; street often not present on resumes) ---
  const loc = d?.location?.parsed || d?.address?.parsed || d?.location || {};
  const physicalAddress = {
    street:  loc.street || loc.streetNumber ? `${loc.streetNumber ?? ''} ${loc.street ?? ''}`.trim() : '',
    city:    loc.city || '',
    state:   loc.stateCode || loc.state || '',
    postal:  loc.postalCode || '',
    country: loc.countryCode || loc.country || ''
  };
  const physicalAddressText = [
    physicalAddress.street,
    [physicalAddress.city, physicalAddress.state].filter(Boolean).join(', '),
    physicalAddress.postal,
    physicalAddress.country
  ].filter(Boolean).join(', ');

  // --- current / most recent job title ---
  const roles = arrayify(d.workExperience).map((r) => {
    const p = r?.parsed || {};
    const start = p?.workExperienceStartDate?.parsed || r?.startDate || null;
    const end   = p?.workExperienceEndDate?.parsed   || r?.endDate   || null;
    const isCurrent = !end || String(end).toLowerCase() === 'present';
    return {
      title: p?.workExperienceJobTitle || r?.jobTitle || '',
      org:   p?.workExperienceOrganization || r?.organization || r?.company || '',
      start, end, isCurrent
    };
  });

  // pick current; else latest by end; else latest by start
  let current = roles.find(x => x.isCurrent);
  if (!current) current = roles
    .slice()
    .sort((a,b) => new Date(b.end||'2100-01-01') - new Date(a.end||'2100-01-01'))[0] || roles[0];

  const currentJobTitle = (current?.title || '').trim();

  return {
    fullLegalName: full,
    email,
    phone,
    physicalAddress: physicalAddressText,     // Step-3 single-line display
    physicalAddressParts: physicalAddress,    // keep parts for Step-4 address split
    linkedin: linkedinUrl,
    currentJobTitle
  };
}
```

export function getStep3Prefill() {

  const li = s3.linkedin || s3.linkedIn || s3.linkedinUrl || '';

  };
}
