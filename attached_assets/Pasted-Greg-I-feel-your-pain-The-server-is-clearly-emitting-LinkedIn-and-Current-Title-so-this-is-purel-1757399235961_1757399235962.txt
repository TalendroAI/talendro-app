## 1) Add a tiny tolerant getter

**client/src/lib/prefill.js**

```js
export function getStep3Prefill() {
  const raw = JSON.parse(localStorage.getItem('resumeData') || 'null');
  const bag = raw?.prefill?.step3 || raw?.data?.prefill?.step3 || {};
  const linkedIn =
    bag.linkedin || bag.linkedIn || bag.linkedinUrl || '';
  const currentJobTitle =
    bag.currentJobTitle || bag.currentTitle || '';
  return { ...bag, linkedIn, currentJobTitle };
}
```

## 2) If you use plain React state (no RHF)

**client/src/app/Onb3.js** (relevant bits)

```jsx
import { useEffect, useMemo, useState } from 'react';
import { getStep3Prefill } from '../lib/prefill';

export default function Onb3() {
  const pre = useMemo(() => getStep3Prefill(), []);
  const [form, setForm] = useState({
    fullLegalName: '',
    email: '',
    phone: '',
    physicalAddress: '',
    linkedin: '',          // <- canonical in UI
    currentJobTitle: ''    // <- canonical in UI
  });

  useEffect(() => {
    setForm(f => ({
      ...f,
      fullLegalName: pre.fullLegalName || '',
      email: pre.email || '',
      phone: pre.phone || '',
      physicalAddress: pre.physicalAddress || '',
      linkedin: pre.linkedIn || '',              // tolerant fill
      currentJobTitle: pre.currentJobTitle || ''
    }));
    console.log('💧Hydrated Step3 from prefill:', pre);
  }, [pre]);

  return (
    <form>
      {/* …other fields… */}
      <label>LinkedIn</label>
      <input
        name="linkedin"
        value={form.linkedin}                  // CONTROLLED value (not placeholder)
        onChange={e => setForm(s => ({ ...s, linkedin: e.target.value }))}
        autoComplete="url"
      />
      <label>Current Job Title</label>
      <input
        name="currentJobTitle"
        value={form.currentJobTitle}
        onChange={e => setForm(s => ({ ...s, currentJobTitle: e.target.value }))}
      />
      {/* … */}
    </form>
  );
}
```

## 3) If you use React Hook Form

**client/src/app/Onb3.js**

```jsx
import { useEffect, useMemo } from 'react';
import { useForm } from 'react-hook-form';
import { getStep3Prefill } from '../lib/prefill';

export default function Onb3() {
  const pre = useMemo(() => getStep3Prefill(), []);
  const { register, reset } = useForm({ defaultValues: {} });

  useEffect(() => {
    reset({
      fullLegalName: pre.fullLegalName || '',
      email: pre.email || '',
      phone: pre.phone || '',
      physicalAddress: pre.physicalAddress || '',
      linkedin: pre.linkedIn || '',                // tolerant
      currentJobTitle: pre.currentJobTitle || ''
    });
    console.log('💧RHF reset with prefill:', pre);
  }, [pre, reset]);

  return (
    <form>
      {/* …other fields… */}
      <label>LinkedIn</label>
      <input {...register('linkedin')} autoComplete="url" />
      <label>Current Job Title</label>
      <input {...register('currentJobTitle')} />
      {/* … */}
    </form>
  );
}
```

> **Gotcha:** don’t mix `defaultValue` and `value` on the same input. With RHF, prefer `reset()` to hydrate.

## 4) Make the server extra-forgiving (optional but recommended)

Where you build `prefill.step3` on the server, emit **all synonyms** so any client variant works:

```js
prefill.step3 = {
  ...prefill.step3,
  linkedin: linkedInUrl || prefill.step3?.linkedin || '',
  linkedIn: linkedInUrl || prefill.step3?.linkedIn || '',
  linkedinUrl: linkedInUrl || prefill.step3?.linkedinUrl || '',
  currentJobTitle: currentTitle || prefill.step3?.currentJobTitle || '',
  currentTitle: currentTitle || prefill.step3?.currentTitle || ''
};
```
