## A) Deterministic “Current/Most-Recent” Job Title

**Why it flips:** Affinda’s `workExperience` can arrive in different shapes and orders. If you just take `[0]` or sort by raw strings, you’ll get inconsistent results.

**Drop-in server patch (recommended)**
*Add this helper where you build `profileDraft`/`prefill.step3` (after you have the Affinda `data`).*

```js
// ── helpers: normalize, detect "current", and pick title ─────────────────────
function normYMD(x) {
  if (!x) return null;
  if (/present|current/i.test(String(x))) return null;
  const s = String(x).trim();
  const m = s.match(/^(\d{4})(?:-(\d{2}))?(?:-(\d{2}))?$/);
  if (m) return `${m[1]}-${m[2] || '12'}-${m[3] || '31'}`;
  const d = new Date(s);
  return isNaN(+d) ? null : d.toISOString().slice(0,10);
}
const ts = (ymd, endBias=false) =>
  !ymd ? (endBias ? Number.POSITIVE_INFINITY : Number.NEGATIVE_INFINITY)
       : +new Date(ymd);

function normalizeWorkArray(d) {
  const src = d?.workExperience ?? [];
  const arr = Array.isArray(src)
    ? src
    : src?.listItem ?? Object.values(src ?? {});
  return (arr || []).map(w => {
    const title = w?.parsed?.workExperienceJobTitle?.value
               || w?.parsed?.workExperienceJobTitle
               || w?.jobTitle?.value || w?.jobTitle
               || w?.position?.value || w?.position || '';
    const startRaw = w?.parsed?.workExperienceStartDate?.parsed
                  || w?.startDate?.parsed || w?.startDate || w?.dates?.start;
    const endRaw   = w?.parsed?.workExperienceEndDate?.parsed
                  || w?.endDate?.parsed || w?.endDate || w?.dates?.end;
    const isCur = w?.end?.isCurrent === true || w?.isCurrent === true
               || /present|current/i.test(String(endRaw || ''));
    return {
      title: String(title || '').trim(),
      start: normYMD(startRaw),
      end:   isCur ? null : normYMD(endRaw),
      isCur
    };
  }).filter(r => r.title);
}

function pickCurrentTitle(d) {
  const rows = normalizeWorkArray(d);
  if (!rows.length) return '';

  // 1) prefer explicitly current; tiebreak = latest start
  const cur = rows.filter(r => r.isCur);
  if (cur.length) {
    cur.sort((a,b) => ts(b.start) - ts(a.start));
    return cur[0].title;
  }
  // 2) else any open-ended; tiebreak = latest start
  const open = rows.filter(r => !r.end);
  if (open.length) {
    open.sort((a,b) => ts(b.start) - ts(a.start));
    return open[0].title;
  }
  // 3) else latest end; tiebreak = latest start
  rows.sort((a,b) => (ts(b.end,true) - ts(a.end,true)) || (ts(b.start) - ts(a.start)));
  return rows[0].title;
}
```

Then set **both synonyms** when building prefill (so any UI binding works):

```js
const currentTitle = pickCurrentTitle(affindaData?.data);
prefill.step3 = {
  ...prefill.step3,
  currentJobTitle: currentTitle || prefill.step3?.currentJobTitle || '',
  currentTitle:    currentTitle || prefill.step3?.currentTitle || '',
};
```
